CREATE OR REPLACE TRIGGER UPD_ITEM_STOCK
AFTER INSERT OR UPDATE OR DELETE ON prod_receive_items
FOR EACH ROW
DECLARE
V_PRODUCT_ID prod_receive_items.PRODUCT_ID%TYPE;
BEGIN
select count(*) into V_PRODUCT_ID from stocks
where PRODUCT_ID=:NEW.PRODUCT_ID;
IF V_PRODUCT_ID = 0 THEN
INSERT INTO STOCKS(PRODUCT_ID,QUANTITY)
VALUES(:NEW.PRODUCT_ID,:NEW.PROD_QTY);
ELSIF V_PRODUCT_ID IS NOT NULL THEN
UPDATE STOCKS
SET QUANTITY=NVL(QUANTITY,0)+:NEW.PROD_QTY
WHERE PRODUCT_ID=:NEW.PRODUCT_ID;
END IF;
EXCEPTION
WHEN NO_DATA_FOUND THEN
NULL;
END;
/

create or replace trigger sales_return_update
after insert or update on SALES_RETURN_ITEMS
for each row
begin
update stocks
set quantity=nvl(quantity,0)+:new.prod_qty
where product_id=:new.product_id;
end;
/

create or replace trigger shipping_return
after insert or update on SHIPPING_RETURN_DETAILS
for each row
begin
update stocks
set quantity=nvl(quantity,0)+:new.SHIPP_RETURN_QTY
where product_id=:new.product_id;
end;
/

create or replace trigger stock_shipping
after insert or update on SHIPPING_DETAILS
for each row
begin
update stocks
set quantity=nvl(quantity,0)-:new.quantity
where product_id=:new.product_id;
end;
/

create or replace trigger pur_return_update
after insert or update on PURCHAGE_RETURN_ITEMS
for each row
begin
update stocks
set quantity=nvl(quantity,0)-:new.prod_qty
where product_id=:new.product_id;
end;
/

CREATE OR REPLACE TRIGGER UPD_DAMAGE_ITEMS
AFTER INSERT OR UPDATE OR DELETE ON PROD_DAMAGE_ITEMS
FOR EACH ROW
DECLARE
V_PRODUCT_ID PROD_DAMAGE_ITEMS.PRODUCT_ID%TYPE;
BEGIN
select count(*) into V_PRODUCT_ID from stocks
where PRODUCT_ID=:NEW.PRODUCT_ID;
IF V_PRODUCT_ID IS NOT NULL THEN
UPDATE STOCKS
SET QUANTITY=NVL(QUANTITY,0)-:NEW.PROD_QTY
WHERE PRODUCT_ID=:NEW.PRODUCT_ID;
END IF;
EXCEPTION
WHEN NO_DATA_FOUND THEN
NULL;
COMMIT;
END;
/
